from pyrogram import Client
user_client = Client("creator_account", api_id, api_hash)
async def crear_grupo_videollamada(creator_name, session_id):
    async with user_client:
        # Crear supergrupo dinámicamente
        chat = await user_client.create_supergroup(
            title=f"VideoLlamada-{creator_name}-{session_id}",
            description="Sesión privada de videollamada"
        )
        
        # Agregar bot como administrador
        await user_client.promote_chat_member(
            chat_id=chat.id,
            user_id=BOT_USER_ID,
            privileges=ChatPrivileges(
                can_manage_video_chats=True,
                can_restrict_members=True,
                can_delete_messages=True
            )
        )
        
        return chat.id
